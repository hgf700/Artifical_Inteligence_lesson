#   bagging kilka modeli uczy się równolegle na różnych losowych próbkach   Parametry:
#   bootstrap=True - losowanie ze zwracaniem
#   oob_score=True - ocena modelu na danych, które nie trafiły do próbki (Out Of Bag)
#   n_jobs=-1 - trenowanie równoległe na wszystkich rdzeniach CPU

bag = BaggingClassifier(
    estimator=DecisionTreeClassifier(random_state=42),
    n_estimators=200,        
    max_samples=0.8,         
    bootstrap=True,          
    n_jobs=-1,               
    random_state=42,
    oob_score=True           
)
bag.fit(X_train, y_train)
